lock steering to up + R(0,0,180).

set seek to 1000.
set pg to 0.1.
set ig to 0.002.
set dg to -0.4.

set scale to 0.1.

until 0 {
set err to seek - altitude.
set p to err * pg.

set i to i + (err * ig).

set d to verticalspeed * dg.


set t to i + p + d.

if t > 1 { set t to 1. }.
if t < 0 { set t to 0. }.

lock throttle to t * scale.

clearscreen.
print "p:" + p.
print "i:" + i.
print "d:" + d.
print t.
}.
